#!/bin/bash

x=1
while [ $x -le 33 ]
do

  echo "-----------------------" >> TimingResults.txt 
  
  
  
  

  echo "" >> TimingResults.txt

  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt
  echo Test Case 100 x 100 : Dens 20 : 100 Iterations >> \
TimingResults.txt

input_100x100_dens_20.txt >> TimingResults.txt

  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt
  echo "" >> TimingResults.txt






  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt
  echo Test Case 1000 x 1000 : Dens 20 : 100 Iterations >> \
TimingResults.txt

input_1000x1000_dens_20.txt >> TimingResults.txt

  echo "" >> TimingResults.txt






  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt
  echo Test Case 2000 x 2000 : Dens 20 : 100 Iterations >> \
TimingResults.txt

  java -Dpj.np=$x GameOfLife 2000 2000 800000 100 \
input_2000x2000_dens_20.txt >> TimingResults.txt

  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt
  echo "" >> TimingResults.txt
  
  
 
 
 
 
 

  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt
  echo Test Case 1000 x 1000 : Dens 40 : 100 Iterations >> \
TimingResults.txt

  java -Dpj.np=$x GameOfLife 1000 1000 200000 100 \
input_1000x1000_dens_40.txt >> TimingResults.txt

  echo "~~~~~~~~~~~~~~~~~~~~~~~" >> TimingResults.txt

TimingResults.txt

input_1000x1000_dens_80.txt >> TimingResults.txt

TimingResults.txt
